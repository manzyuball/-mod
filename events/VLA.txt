# Tizard mission event for USA
country_event = {
	id = VLA.1
	title = VLA.1.t
	desc = VLA.1.d
	picture = GFX_report_event_journalists_speech
		fire_only_once = yes
	
	trigger = {
		focus_progress = {
			focus = VLA_se1
			progress > 0.1
			}
		tag = VLA
	}
	
	option = {
		name = VLA.1.a
		ai_chance = { factor = 50 }
		custom_effect_tooltip = VLA_teise
		add_to_variable = {
			var = VLA.VLA_teise_h

		}
	}
	
	option = {
		name = VLA.1.b
		ai_chance = { factor = 50 }
		custom_effect_tooltip = VLA_gunk

}
