add_namespace = super

country_event = {	#USA_CIV
	id = super.1
	title = super.1.t
	desc = super.1.d
	picture = GFX_report_event_diplomatic_message

	is_triggered_only = no
	
	fire_only_once = yes
	
	hidden = yes
	
	trigger = {
		AND = {
			CSA = { has_war = yes }
			NOT = {
				has_global_flag = Super_WW2
			}
		}
	}
	

	option = {
		name = super.1.a
		
        set_global_flag = USA_CIV
		set_global_flag = Super_Event_Visible
		play_song = "America_civil_war"
		
	}
}
